**Step 3: Create `BeatStep/Views/Player/CollapsiblePlayerView.swift`.**

This is a new wrapper view. Full implementation spec:

```swift
import SwiftUI

struct CollapsiblePlayerView: View {
    @AppStorage("playerCollapsed") private var isCollapsed = false  // D-10: default expanded, D-11: persists
    @State private var dragOffset: CGFloat = 0
    @State private var isDragging = false
    @State private var hasPassedThreshold = false  // Prevents multiple haptic fires

    private let expandedHeight = ComponentSize.miniPlayerHeight        // 64
    private let collapsedHeight = ComponentSize.miniPlayerCollapsedHeight  // 20
    private let dragThreshold: CGFloat = 40  // UI-SPEC: half the height difference

    /// Base height at rest (no drag in progress)
    private var baseHeight: CGFloat {
        isCollapsed ? collapsedHeight : expandedHeight
    }

    /// Current height accounting for active drag
    private var currentHeight: CGFloat {
        let offset = isCollapsed ? -dragOffset : dragOffset
        // Negative offset = user dragging toward opposite state
        let target = baseHeight + offset
        return min(max(target, collapsedHeight), expandedHeight)
    }

    /// 0.0 = fully collapsed, 1.0 = fully expanded
    private var expandProgress: CGFloat {
        let range = expandedHeight - collapsedHeight
        guard range > 0 else { return 0 }
        return (currentHeight - collapsedHeight) / range
    }

    var body: some View {
        ZStack {
            // Expanded content layer -- fades out as player collapses (D-08)
            MiniPlayerView()
                .opacity(expandProgress)
                .allowsHitTesting(expandProgress > 0.5)  // Disable button hits when mostly collapsed

            // Collapsed pill handle -- fades in as player collapses (D-09 cross-fade)
            Capsule()
                .fill(Color.textTertiary)  // white 35%
                .frame(width: ComponentSize.dragHandleWidth,     // 36pt
                       height: ComponentSize.dragHandleHeight)   // 4pt
                .opacity(1 - expandProgress)
        }
        .frame(height: currentHeight)
        .frame(maxWidth: .infinity)
        .background(
            Rectangle()
                .fill(.ultraThinMaterial)  // D-06: same material both states
                .shadow(color: .black.opacity(0.1), radius: 4, y: -2)  // D-07: keep shadow
        )
        .contentShape(Rectangle())  // Full area tappable/draggable
        .gesture(dragGesture)
        .onTapGesture { toggleState() }  // D-01/D-02: tap to toggle
        .accessibilityLabel(isCollapsed
            ? "Music player, collapsed. Tap or swipe up to expand."
            : "Music player. Swipe down to minimize.")
    }

    // MARK: - Gestures

    private var dragGesture: some Gesture {
        DragGesture(minimumDistance: 8)  // UI-SPEC: 8pt minimum
            .onChanged { value in
                // Only commit to vertical drags (Pitfall 1: direction ambiguity)
                guard abs(value.translation.height) > abs(value.translation.width) else { return }
                isDragging = true
                dragOffset = value.translation.height

                // D-04: Haptic at threshold crossing (Pitfall 5: debounced)
                let crossed = abs(value.translation.height) > dragThreshold
                if crossed && !hasPassedThreshold {
                    BSHaptics.light()
                    hasPassedThreshold = true
                } else if !crossed && hasPassedThreshold {
                    hasPassedThreshold = false
                }
            }
            .onEnded { value in
                isDragging = false
                hasPassedThreshold = false

                let shouldToggle = abs(value.translation.height) > dragThreshold
                if shouldToggle {
                    // Check drag direction matches expected state change
                    let isDraggingDown = value.translation.height > 0
                    if (isDraggingDown && !isCollapsed) || (!isDraggingDown && isCollapsed) {
                        withAnimation(BSAnimation.smooth) {
                            isCollapsed.toggle()
                        }
                    }
                }
                withAnimation(BSAnimation.smooth) {
                    dragOffset = 0
                }
            }
    }

    private func toggleState() {
        BSHaptics.light()  // D-04
        withAnimation(BSAnimation.smooth) {
            isCollapsed.toggle()
        }
    }
}
```

Key decisions implemented:
- D-01/D-02: Swipe + tap to collapse/expand
- D-03: Interactive drag -- height follows finger via `currentHeight` computed from `dragOffset`
- D-04: BSHaptics.light() on threshold crossing with debounce flag
- D-05: Centered capsule pill, 36x4pt, no text
- D-06: Full-width ultraThinMaterial background in both states, 20pt collapsed height
- D-07: Same shadow in both states
- D-08/D-09: Cross-fade -- content opacity = expandProgress, pill opacity = 1 - expandProgress
- D-10: Default `isCollapsed = false` (expanded)
- D-11: `@AppStorage("playerCollapsed")` persists across restarts

Anti-patterns avoided:
- No GestureState (uses @State + manual reset per project convention)
- No if/else view switching (single ZStack with opacity layers)
- No raw UIKit haptics (uses BSHaptics token)
- Direction guard prevents horizontal scroll conflicts (Pitfall 1)
- hasPassedThreshold flag prevents multiple haptic fires (Pitfall 5)
